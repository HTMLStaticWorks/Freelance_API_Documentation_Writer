REST API Reference

Detailed endpoint documentation with request/response examples.

GET

Retrieve Account Balance

Production Ready
/v1/account/balance

Returns the current balance and transaction history for the authenticated user. This endpoint requires a valid API Key with read:account permissions.

Query Parameters
Parameter Type Required
currency string Required
The ISO 4217 currency code (e.g. USD, EUR).
limit integer Optional
Number of transactions to return (Default: 10, Max: 100).
Response Example
200 OK 
{
  "status": "success",
  "data": {
    "balance": 1250.75,
    "currency": "USD",
    "updated_at": "2026-05-13T10:00:00Z"
  }
}

SDK Implementation

Pre-built libraries for your favorite programming languages.

const forge = require('apiforge-node')('your_api_key');

async function getBalance() {
  try {
    const balance = await forge.account.getBalance({
      currency: 'USD'
    });
    console.log(`Current Balance: ${balance.amount} ${balance.currency}`);
  } catch (error) {
    console.error('API Error:', error.message);
  }
}
from apiforge import Client

client = Client(api_key="your_api_key")

def get_balance():
    try:
        balance = client.account.get_balance(currency="USD")
        print(f"Current Balance: {balance['amount']} {balance['currency']}")
    except Exception as e:
        print(f"API Error: {str(e)}")
package main

import (
    "fmt"
    "github.com/apiforge/apiforge-go"
)

func main() {
    client := apiforge.NewClient("your_api_key")
    
    balance, err := client.Account.GetBalance("USD")
    if err != nil {
        fmt.Printf("API Error: %v\n", err)
        return
    }
    
    fmt.Printf("Current Balance: %f %s\n", balance.Amount, balance.Currency)
}

Webhook Events

Real-time event notifications delivered via HTTP POST.

payment.succeeded

This event is triggered immediately after a payment transaction is successfully authorized and captured by the banking network.

Event Metadata
  • Retry Logic
    Retries up to 5 times with exponential backoff.
  • Signature Verification
    Includes X-Forge-Signature for security.
Payload: JSON 
{
  "id": "evt_987654321",
  "object": "event",
  "created": 1652432400,
